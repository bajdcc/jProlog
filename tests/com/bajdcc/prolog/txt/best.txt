/** 最佳选手
 * 斯科特先生、他的妹妹、他的儿子、他的女儿都是网球选手。关于这四人有以下的情况:
 * (1) 最佳选手的的孪生同胞与最差选手的性别相同。
 * (2) 最佳选手与最差选手的年龄相同
 * 请问谁是最佳选手
 */

//变量定义部分
所有人{} = {"斯科特","妹妹","儿子","女儿"};
性别{} = {"男","女"};
年龄{} = {1,2,3,4}; //本题不关心年龄的实际数值，只关心大小，因此四个人年龄的取值是1,2,3,4足够了
年龄 year(所有人);
性别 sex(所有人);
所有人 同胞(所有人);
所有人 最差,最佳;

//条件定义部分
sex("斯科特") = "男";
sex("妹妹") = "女";
sex("儿子") = "男";
sex("女儿") = "女";
同胞("斯科特") = "妹妹";
同胞("妹妹") = "斯科特";
同胞("儿子") = "女儿";
同胞("女儿") = "儿子";
最佳 != 最差;
year("斯科特") > year("儿子");
year("斯科特") > year("女儿");
year(最佳) = year(同胞(最佳)); //(1)最佳选手的的孪生同胞
sex(同胞(最佳)) != sex(最差);  //(1)
year(最差) = year(最佳);       //(2)

输出("最佳选手是: ", 最佳);
输出("最差选手是: ", 最差);
